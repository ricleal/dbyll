#include <iostream>
using namespace std;
 
int main(int argc, char* argv[])
{
    int i = 5;
    int *p = &i;
        int &r = i;
 
    cout << " i =5 ; *p = &i ; &r = i " << endl;
    cout << "Result: " << endl;
    cout << " i =" <<  i << endl;
    cout << "&i =" << &i << endl;
    cout << "*p =" << *p << endl;
    cout << " p =" << p << endl;
    cout << "&p =" << &p << endl;
    cout << " r =" <<  r << endl;
    cout << "&r =" << &r << endl;
 
        cout << "Let's change the value of i: "  << endl;
    i = 6;
    cout << " i =6 ; *p = &i ; &r = i " << endl;
    cout << "Result: " << endl;
    cout << " i =" <<  i << endl;
    cout << "&i =" << &i << endl;
    cout << "*p =" << *p << endl;
    cout << " p =" << p << endl;
    cout << "&p =" << &p << endl;
    cout << " r =" <<  r << endl;
    cout << "&r =" << &r << endl;
 
    return 0;
}
